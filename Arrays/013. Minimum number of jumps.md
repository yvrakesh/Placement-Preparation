Efficient Approach:

Time Complexity : O(n)

Space Complexity : O(1)

    class Solution {
    public:
        bool canJump(vector<int>& nums) {
            int n = nums.size();
            int reachable_pos = 0;
            for(int i=0;i<n;i++){
                if(reachable_pos < i)
                    return false;
                reachable_pos = max(reachable_pos,i+nums[i]);
                if(reachable_pos > n-2)
                    return true;
            }
            return true;
        }
    };
